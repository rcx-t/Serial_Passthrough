30void uart0_init(uint32_t baud_rate) {

57void uart3_init(uint32_t baud_rate) { // USB
58 uint16_t baud = calc_baud(baud_rate, F_CPU); // baud rate
59 USART3.BAUD = baud;
60 USART3.CTRLB = USART_RXEN_bm | USART_TXEN_bm; // Enable Rx in Tx
61 USART3.CTRLC = USART_CHSIZE0_bm | USART_CHSIZE1_bm; // 8-bits
